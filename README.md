
```

```

```sql
/* 创建帐号权限 */

grant all privileges on test_www_db.* to 'test_www'@'%';
flush privileges;


/* 操作 数据库 */

/* 创建 一个名称为 tb_user 的表*/
CREATE TABLE tb_user( 

	id int(32) PRIMARY KEY AUTO_INCREMENT,
	username varchar(32),
```
